# Firebase Firestore Security Rules for Blog System
# More permissive rules for testing

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    # Allow all read/write operations for authenticated users
    match /{document=**} {
      allow read, write: if request.auth != null;
    }
    
    # Allow public read access for blogs
    match /blogs/{blogId} {
      allow read: if resource.data.status == 'published';
      allow write, delete: if request.auth != null;
    }
    
    match /blogs {
      allow list: if true;
      allow get: if true;
    }
    
    # Users collection - authenticated access
    match /users/{userId} {
      allow read, write, delete: if request.auth != null;
    }
    
    match /users {
      allow list: if request.auth != null;
    }
  }
}

# Rules for storage (for image uploads)
rules_version = '2';
service firebase.storage {
  match /b/{allPaths=**} {
    allow read, write: if request.auth != null;
  }
}
